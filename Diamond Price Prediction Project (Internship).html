def detect_outliers (data, column):
    q1 = dp[column].quantile (.25)
    q3= dp[column].quantile (.75)
    IQR = q3-q1
    lower_bound = q1-(1.5*IQR)
    upper_bound = q3+ (1.5*IQR)
    ls = dp.index [ (dp [column] < lower_bound) | (dp [column] > upper_bound) ]
    return ls
